Shell执行任务时的loading效果

脚本

# loading状态方法
function spinner {
	local pid=$1
	local delay=0.15
	local spinstr='|/-\'
		while [ "$(ps a | awk '{print $1}' | grep $pid)" ]; do
			local temp=${spinstr#?}
			printf " [%c]  " "$spinstr"
			local spinstr=$temp${spinstr%"$temp"}
			sleep $delay
			printf "\b\b\b\b\b\b"
		done
	printf "    \b\b\b\b"
}

# 模拟耗时的任务
function execute() {
    sleep 5
}

echo -n "Loading..."

# 后台执行耗时的任务
execute &
# 启动loading效果
spinner "$!"

echo ""
